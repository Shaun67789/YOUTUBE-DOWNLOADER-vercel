GET /api/download?url={youtube_url}

Download

Download a YouTube video or audio stream directly. Returns the file stream.

ParameterTypeRequiredDescription
urlstringYesFull YouTube URL (e.g. https://youtu.be/VIDEO_ID)
cURL
Python
JavaScript
# Download a video
curl -L -o video.mp4 "https://your-app.vercel.app/api/download?url=https://youtu.be/D8YEkMjNumE"
# Python
import requests

url = "https://your-app.vercel.app/api/download"
params = {"url": "https://youtu.be/D8YEkMjNumE"}
r = requests.get(url, params=params, stream=True)
with open("video.mp4", "wb") as f:
    for chunk in r.iter_content(1024):
        f.write(chunk)
// JavaScript (Node.js)
const fs = require("fs");
const https = require("https");

const url = "https://your-app.vercel.app/api/download?url=" +
  encodeURIComponent("https://youtu.be/D8YEkMjNumE");

https.get(url, (res) => {
  const file = fs.createWriteStream("video.mp4");
  res.pipe(file);
});
GET /api/info?url={youtube_url}

Video Info

Get video metadata (title, thumbnail, duration, etc.) with owner info and download link.

ParameterTypeRequiredDescription
urlstringYesFull YouTube URL
cURL
Python
JavaScript
# Get video info
curl "https://your-app.vercel.app/api/info?url=https://youtu.be/D8YEkMjNumE" | jq .
# Python
import requests, json

url = "https://your-app.vercel.app/api/info"
params = {"url": "https://youtu.be/D8YEkMjNumE"}
r = requests.get(url, params=params)
data = r.json()
print(json.dumps(data, indent=2))
// JavaScript
const res = await fetch(
  "/api/info?url=" + encodeURIComponent("https://youtu.be/D8YEkMjNumE")
);
const data = await res.json();
console.log(data);
